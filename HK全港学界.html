function render() {

  const tbody = document.getElementById("tbody");

  tbody.innerHTML = "";

  results.forEach(r => {

    const tr = document.createElement("tr");

    tr.innerHTML = `<td>${r.y}</td><td>${r.g}</td><td>${r.e}</td><td>${r.n}</td><td>${r.d}</td><td>${r.t}</td><td>${r.r}</td>`;

    tbody.appendChild(tr);

  });

}


function filter() {

  const name = document.getElementById("name").value.toLowerCase();

  const dist = document.getElementById("district").value.toLowerCase();

  const yr = document.getElementById("year").value;

  const rows = document.querySelectorAll("#tbody tr");

  rows.forEach(row => {

    const n = row.cells[3].textContent.toLowerCase();

    const d = row.cells[4].textContent.toLowerCase();

    const y = row.cells[0].textContent;

    row.style.display = (n.includes(name) && d.includes(dist) && (!yr || y === yr)) ? "" : "none";

  });

}
